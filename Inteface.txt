import streamlit as st
import requests

# URL da sua API FastAPI
API_URL = "http://127.0.0.1:8000/chat"

st.set_page_config(page_title="Agente de Projetos", page_icon="ğŸ¤–", layout="wide")
st.title("ğŸ’¬ Chat com Agente de Projetos")

# Inicializa histÃ³rico na sessÃ£o
if "messages" not in st.session_state:
    st.session_state["messages"] = []

# Exibe histÃ³rico
for msg in st.session_state["messages"]:
    if msg["role"] == "user":
        st.chat_message("user").markdown(msg["content"])
    else:
        st.chat_message("assistant").markdown(msg["content"])

# Caixa de entrada
if prompt := st.chat_input("Digite sua pergunta..."):
    # Adiciona no histÃ³rico
    st.session_state["messages"].append({"role": "user", "content": prompt})
    st.chat_message("user").markdown(prompt)

    # Envia para API
    response = requests.post(API_URL, json={"pergunta": prompt})
    resposta = response.json()["resposta"]

    # Mostra resposta
    st.session_state["messages"].append({"role": "assistant", "content": resposta})
    st.chat_message("assistant").markdown(resposta)
