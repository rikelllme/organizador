    <script>
        document.addEventListener('DOMContentLoaded', () => {
            function getWeekNumber(d) {
                d = new Date(Date.UTC(d.getFullYear(), d.getMonth(), d.getDate()));
                d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay() || 7));
                const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
                const weekNo = Math.ceil((((d - yearStart) / 86400000) + 1) / 7);
                return weekNo;
            }

            const currentWeek = getWeekNumber(new Date());
            const week1 = currentWeek;
            const week2 = currentWeek + 1;
            const week3 = currentWeek + 2;

            document.getElementById('week-head-1').textContent = `Semana ${week1}`;
            document.getElementById('week-head-2').textContent = `Semana ${week2}`;
            document.getElementById('week-head-3').textContent = `Semana ${week3}`;

            document.getElementById('week-range').textContent = `(W${week1} a W${week3})`;
        });
